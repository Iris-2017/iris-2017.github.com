第二步： 创建消息类型

定义消息类型保证不重复即可，格式例如：@cmb_test@

第三步： 引入 SDK 并创建实例初始化

引入 yuntu JavaScript SDK 你需要用你的证书信息 或 用我们做演示的证书 来实例化一个websocket对象。 接着，注册到服务器，并连接到服务器。 

  1. <html xmlns="http://www.w3.org/1999/xhtml">
  2.   <head>
  3.   <meta http-equiv="Content-Type" content="text/html; charset=utf-8"/>
  4.   <script type="text/javascript" src="cmb.1.0.js"></script>
  5.   <title>云途demo</title>
  6.   <script>
  7.   var sck = null;
  8.   var cert = {
  9.   "check" : "6a01f4a52c71c0f018d45d3ea8e307fe",
  10.   "info" : {
  11.   "cmb" : {
  12.   "id" : "{61C4DA8C-2BEA-4103-942A-A074F7CC6A6B}",
  13.   "ip" : "127.0.0.1",
  14.   "notify" : "@cmb_test@",
  15.   "subscribe" : "@cmb_test@",
  16.   "user" : 5,
  17.   "valid" : "2016-12-31"
  18.   }
  19.   }
  20.   };
  21.   function init(){
  22.   sck = new cmb.ws.websocket('',cert,
  23.   function(event){
  24.   document.getElementById("responses").innerHTML = "初始化成功!";
  25.   sck.register(cert);
  26.   },function(event){
  27.   //收到消息后处理
  28.   document.getElementById("responses").innerHTML += "<br/>收到消息：" + event.data;
  29.   },function(event){
  30.   //关闭连接后处理
  31.   document.getElementById("responses").innerHTML += "<br/>" + event.data;
  32.   },function(event){
  33.   //出现错误后处理
  34.   document.getElementById("responses").innerHTML += "<br/>" + event.data;
  35.   });
  36.   }
  37.   /**
  38.   * 发送消息
  39.   * @param msg
  40.   */
  41.   function send_msg(){
  42.   var note = document.getElementById("msg").value;
  43.   var msg = {
  44.   "note":note
  45.   };
  46.   sck.uploadmsg(msg,"@cmb_test@");
  47.   document.getElementById("send").innerHTML += "<br/>发送的消息：" + JSON.stringify(msg);
  48.   }
  49.   function close()
  50.   {
  51.   sck.close();
  52.   };
  53.   function getmsg(){
  54.   if(sck){
  55.   sck.subscribe("@cmb_test@");
  56.   }
  57.   }
  58.   </script>
  59.   </head>
  60.   <body>
  61.   <button class="btn btn-success" onclick="init();">初始化</button>
  62.   <button class="btn btn-success" onclick="getmsg();">订阅消息</buttonn>
  63.   <button class="btn btn-danger" onclick="close();">断开连接</button>
  64.   <span class="input-group-addon">发送的消息</span>
  65.   <input type="text" name="msg" id="msg" value="这是一个测试消息" class="form-control" placeholder="用户名">
  66.   <button class="btn btn-default"  onclick="send_msg();">发送消息</button>
  67.   <div id="send" style="height: 100px;border:1px solid #F00;OVERFLOW-Y: auto; OVERFLOW-X:hidden;"></div>
  68.   <div id="responses" style="height: 200px;border:1px solid #F00;OVERFLOW-Y: auto; OVERFLOW-X:hidden;"></div>
  69.   </body>
  70.   </html>
